017package org.apache.activemq.transport.reliable;
018
019import java.io.IOException;
020
021/**
022 * Used by a {@link ReplayBuffer} to replay buffers back over an unreliable transport
023 * 
024 * 
025 */
026public interface Replayer {
027
028    /**
029     * Sends the given buffer back to the transport
030     * if the buffer could be found - otherwise maybe send some kind
031     * of exception
032     * 
033     * @param commandId the command ID
034     * @param buffer the buffer to be sent - or null if the buffer no longer exists in the buffer
035     */
036    void sendBuffer(int commandId, Object buffer) throws IOException;
037}
