017package org.apache.activemq.store;
018
019import org.apache.activemq.util.ByteSequence;
020
021public class PListEntry {
022
023    private final ByteSequence byteSequence;
024    private final String entry;
025    private final Object locator;
026
027    public PListEntry(String entry, ByteSequence bs, Object locator) {
028        this.entry = entry;
029        this.byteSequence = bs;
030        this.locator = locator;
031    }
032
033    public ByteSequence getByteSequence() {
034        return this.byteSequence;
035    }
036
037    public String getId() {
038        return this.entry;
039    }
040
041    public Object getLocator() {
042        return locator;
043    }
044
045    public PListEntry copy() {
046        return new PListEntry(this.entry, this.byteSequence, locator);
047    }
048}
