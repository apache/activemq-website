017package org.apache.activemq.memory;
018
019import java.util.LinkedHashMap;
020import java.util.Map;
021
022/**
023 * A simple least-recently-used cache of a fixed size.
024 * 
025 * 
026 */
027public class LRUMap<K,V> extends LinkedHashMap<K,V>{
028
029    protected static final float DEFAULT_LOAD_FACTOR = (float) 0.75;
030    protected static final int DEFAULT_INITIAL_CAPACITY = 5000;
031    private static final long serialVersionUID = -9179676638408888162L;
032
033    private int maximumSize;
034
035    public LRUMap(int maximumSize) {
036        this(DEFAULT_INITIAL_CAPACITY, DEFAULT_LOAD_FACTOR, true, maximumSize);
037    }
038
039    public LRUMap(int maximumSize, boolean accessOrder) {
040        this(DEFAULT_INITIAL_CAPACITY, DEFAULT_LOAD_FACTOR, accessOrder, maximumSize);
041    }
042
043    public LRUMap(int initialCapacity, float loadFactor, boolean accessOrder, int maximumSize) {
044        super(initialCapacity, loadFactor, accessOrder);
045        this.maximumSize = maximumSize;
046    }
047
048    protected boolean removeEldestEntry(Map.Entry<K,V> eldest) {
049        return size() > maximumSize;
050    }
051}
